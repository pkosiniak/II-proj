/app     # Otwierając apkę
   /login      # Muszę się zalogowac
      ->response true      # Jeśli się zalogowałem to
         ../Bills                # Chce przejśc do listy rachunkow
         ../Clients              # Chce przejśc do listy klientów
      ->response false     # Jeśli się nie zalogowałem to
         ./login                 # Chcę zostac i spróbowac jeszcze raz

   /Clients    # Chcę zobaczyć liste klientów
      ->addClient     # Chcę dodać klienta
      /{id}          # Chcę wybrać klienta
         ->deleteClient        # Chcę usunąć klienta
         ?->readClientData      # Chcę zobaczyć dane klienta
            ?/contactsdata
         ->upadteClientData    # Chcę zaktualizować dane klienta
         ->readClientBills     # Chcę zobaczyć wszystkie rachunki klienta
            /Bills
               /{id}          # Chcę sprawdzić historię konkretnego rachunku klienta
                  ?/transactions

            /transactions


   /Bills      # Chcę zobaczyć historię rachunków

      ->addBill   # Chcę dodać nowy rachunek
         ./{id}    // Przejdz do nowo utworzonego rachunku

      ~->readOpened     # Chcę zobaczyć otwarte rachunki ?

      /{id}       # Chcę zobaczyć konkretny rachunek

         /Transactions     # Chcę zobaczyć listę transakji
            /{id}
               ->createTransaction     # Chcę dodać nową transakcje
               ->readTransaction       # Chcę wyświetlić konkretną transakcje
               ?->updateTransaction     # Chcę oznaczyć transakję jako błędną ?
                  ?./->createTransaction     // Przejdz do tworzenia nowej transakcji ? oznaczonej jako poprawiona ?

         ?->updateBill      # Chcę aktualizować stan konta ?

         ~->readClientBills    # Chcę zobaczyć klienta, do którego należy rachunek

         ->closeBill       # Chcę zamknąć rachunek
            ./{id}/->createTransaction      // Przejdz do i dodaj transakcje na kwotę (-)salda rachunku

      ~->closed     # Chcę zobaczyć zamknięte rachunki?
